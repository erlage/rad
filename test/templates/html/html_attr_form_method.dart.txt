test('should set form attribute "method"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), method: FormMethod.get),
            __WidgetClass__(key: GlobalKey('el-2'), method: FormMethod.post),
        ],
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(
        element1.getAttribute('method'),
        equals(fnMapFormMethod(FormMethod.get)),
    );
    expect(
        element2.getAttribute('method'),
        equals(fnMapFormMethod(FormMethod.post)),
    );
});

test('should update form attribute "method"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), method: FormMethod.get),
            __WidgetClass__(key: GlobalKey('el-2'), method: FormMethod.post),
            __WidgetClass__(key: GlobalKey('el-3'), method: FormMethod.get),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1')),
            __WidgetClass__(key: GlobalKey('el-2'), method: null),
            __WidgetClass__(key: GlobalKey('el-3'), method: FormMethod.post),
        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');
    var element3 = app!.elementByGlobalKey('el-3');

    expect(
        element1.getAttribute('method'),
        equals(null),
    );
    expect(
        element2.getAttribute('method'),
        equals(null),
    );
    expect(
        element3.getAttribute('method'),
        equals(fnMapFormMethod(FormMethod.post)),
    );
});
