test('should set attribute "value"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), value: 10),
            __WidgetClass__(key: GlobalKey('el-2'), value: 0),
        ],
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('value'), equals('10'));
    expect(element2.getAttribute('value'), equals('0'));
});

test('should update attribute "value"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), value: 10),
            __WidgetClass__(key: GlobalKey('el-2'), value: 10),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), value: 20),
            __WidgetClass__(key: GlobalKey('el-2'), value: 20),
        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('value'), equals('20'));
    expect(element2.getAttribute('value'), equals('20'));
});

test('should clear attribute "value"', () async {

    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1')),

        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('value'), equals(null));
    expect(element2.getAttribute('value'), equals(null));
});
