test('should set attribute "src"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), src: 'some-src'),
            __WidgetClass__(key: GlobalKey('el-2'), src: 'another-src'),
        ],
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('src'), equals('some-src'));
    expect(element2.getAttribute('src'), equals('another-src'));
}__Skip__);

test('should update attribute "src"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), src: 'some-src'),
            __WidgetClass__(key: GlobalKey('el-2'), src: 'another-src'),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), src: 'updated-src'),
            __WidgetClass__(key: GlobalKey('el-2'), src: 'another-src'),

        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('src'), equals('updated-src'));

}__Skip__);

test('should clear attribute "src"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1')),
            __WidgetClass__(key: GlobalKey('el-2'), src: 'another-src'),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1')),
            __WidgetClass__(key: GlobalKey('el-2')),
        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('src'), equals(null));
    expect(element2.getAttribute('src'), equals(null));
});

test('should clear attribute "src" if updated value is null', () async {
   await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), src: 'some-src'),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), src: null),
        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');

    expect(element1.getAttribute('src'), equals(null));
});
