test('should set attribute "for"', () async {
    await app!.buildChildren(
    widgets: [
        __WidgetClass__(key: GlobalKey('el-1'), forAttribute: 'some-for'),
        __WidgetClass__(key: GlobalKey('el-2'), forAttribute: 'another-for'),
    ],
    parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('for'), equals('some-for'));
    expect(element2.getAttribute('for'), equals('another-for'));
});

test('should update attribute "for"', () async {
    await app!.buildChildren(
    widgets: [
        __WidgetClass__(key: GlobalKey('el-1'), forAttribute: 'some-for'),
        __WidgetClass__(key: GlobalKey('el-2'), forAttribute: 'another-for'),
    ],
    parentContext: app!.appContext,
    );

    await app!.updateChildren(
    widgets: [
        __WidgetClass__(key: GlobalKey('el-1'), forAttribute: 'updated-for'),
        __WidgetClass__(key: GlobalKey('el-2'), forAttribute: 'another-for'),
    ],
    updateType: UpdateType.setState,
    parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('for'), equals('updated-for'));
    expect(element2.getAttribute('for'), equals('another-for'));
});

test('should clear attribute "for"', () async {
    await app!.buildChildren(
    widgets: [
        __WidgetClass__(key: GlobalKey('el-1')),
        __WidgetClass__(key: GlobalKey('el-2'), forAttribute: 'another-for'),
    ],
    parentContext: app!.appContext,
    );

    await app!.updateChildren(
    widgets: [
        __WidgetClass__(key: GlobalKey('el-1')),
        __WidgetClass__(key: GlobalKey('el-2')),
    ],
    updateType: UpdateType.setState,
    parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');
    var element2 = app!.elementByGlobalKey('el-2');

    expect(element1.getAttribute('for'), equals(null));
    expect(element2.getAttribute('for'), equals(null));
});

test('should clear attribute "for" if updated value is null', () async {
   await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), forAttribute: 'some-for'),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), forAttribute: null),
        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var element1 = app!.elementByGlobalKey('el-1');

    expect(element1.getAttribute('for'), equals(null));
});
