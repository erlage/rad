test('should set attribute "selected" only if its true', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), selected: false),
            __WidgetClass__(key: GlobalKey('el-2'), selected: null),
            __WidgetClass__(key: GlobalKey('el-3'), selected: true),
        ],
        parentContext: app!.appContext,
    );

    var domNode1 = app!.domNodeByGlobalKey('el-1');
    var domNode2 = app!.domNodeByGlobalKey('el-2');
    var domNode3 = app!.domNodeByGlobalKey('el-3');

    expect(domNode1.getAttribute('selected'), equals(null));
    expect(domNode2.getAttribute('selected'), equals(null));
    expect(domNode3.getAttribute('selected'), equals('true'));

test('should clear attribute "selected" if updated value is not true', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), selected: true),
            __WidgetClass__(key: GlobalKey('el-2'), selected: true),
            __WidgetClass__(key: GlobalKey('el-3'), selected: true),
            __WidgetClass__(key: GlobalKey('el-4'), selected: true),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), selected: true),
            __WidgetClass__(key: GlobalKey('el-2'), selected: false),
            __WidgetClass__(key: GlobalKey('el-3'), selected: null),
            __WidgetClass__(key: GlobalKey('el-4')),
        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var domNode1 = app!.domNodeByGlobalKey('el-1');
    var domNode2 = app!.domNodeByGlobalKey('el-2');
    var domNode3 = app!.domNodeByGlobalKey('el-3');
    var domNode4 = app!.domNodeByGlobalKey('el-4');

    expect(domNode1.getAttribute('selected'), equals('true'));
    expect(domNode2.getAttribute('selected'), equals(null));
    expect(domNode3.getAttribute('selected'), equals(null));
    expect(domNode4.getAttribute('selected'), equals(null));
});
