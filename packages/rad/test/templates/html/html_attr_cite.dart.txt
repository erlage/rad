    await app!.buildChildren(
    widgets: [
        __WidgetClass__(key: GlobalKey('el-1'), cite: 'some-cite'),
        __WidgetClass__(key: GlobalKey('el-2'), cite: 'another-cite'),

    parentContext: app!.appContext,
    );
