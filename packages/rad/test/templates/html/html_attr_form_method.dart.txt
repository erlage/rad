test('should set form attribute "method"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), method: FormMethodType.get),
            __WidgetClass__(key: GlobalKey('el-2'), method: FormMethodType.post),
        ],
        parentContext: app!.appContext,
    );

    var domNode1 = app!.domNodeByGlobalKey('el-1');
    var domNode2 = app!.domNodeByGlobalKey('el-2');

    expect(
        domNode1.getAttribute('method'),
        equals(FormMethodType.get.nativeName),
    );
    expect(
        domNode2.getAttribute('method'),
        equals(FormMethodType.post.nativeName),
    );
});

test('should update form attribute "method"', () async {
    await app!.buildChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1'), method: FormMethodType.get),
            __WidgetClass__(key: GlobalKey('el-2'), method: FormMethodType.post),
            __WidgetClass__(key: GlobalKey('el-3'), method: FormMethodType.get),
        ],
        parentContext: app!.appContext,
    );

    await app!.updateChildren(
        widgets: [
            __WidgetClass__(key: GlobalKey('el-1')),
            __WidgetClass__(key: GlobalKey('el-2'), method: null),
            __WidgetClass__(key: GlobalKey('el-3'), method: FormMethodType.post),
        ],
        updateType: UpdateType.setState,
        parentContext: app!.appContext,
    );

    var domNode1 = app!.domNodeByGlobalKey('el-1');
    var domNode2 = app!.domNodeByGlobalKey('el-2');
    var domNode3 = app!.domNodeByGlobalKey('el-3');

    expect(
        domNode1.getAttribute('method'),
        equals(null),
    );
    expect(
        domNode2.getAttribute('method'),
        equals(null),
    );
    expect(
        domNode3.getAttribute('method'),
        equals(FormMethodType.post.nativeName),
    );
});
